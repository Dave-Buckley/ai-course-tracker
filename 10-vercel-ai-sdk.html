generateText — non-streaming

Create src/app/api/generate/route.ts

import { generateText } from "ai";

export async function POST(req: Request) {
  const { prompt } = await req.json();
  const { text } = await generateText({
    model: "google/gemini-2.5-flash-lite",
    prompt,
  });
  return Response.json({ text });
}

Test in PowerShell:

curl -X POST http://localhost:3000/api/generate -H "Content-Type: application/json" -d "{\"prompt\":\"Say hi in one sentence\"}"
Worked? JSON response with the LLM's text.

import { streamText } from "ai";

export async function POST(req: Request) {
  const { messages } = await req.json();
  const result = streamText({
    model: "google/gemini-2.5-flash-lite",

Frontend chat with useChat

"use client";

import { useChat } from "ai/react";

export default function Chat() {
  const { messages, input, handleInputChange, handleSubmit } = useChat();
  return (
    <div className="max-w-2xl mx-auto p-6">
      {messages.map((m) => (
        <div key={m.id} className="mb-3">
          <b>{m.role}:</b> {m.content}
        </div>
      ))}

  );
}

Structured output with Zod

Create src/app/api/structured/route.ts

import { generateObject } from "ai";
import { z } from "zod";

export async function POST(req: Request) {
  const { text } = await req.json();
  const { object } = await generateObject({
    model: "google/gemini-2.5-flash-lite",
    schema: z.object({
      summary: z.string(),
      bullets: z.array(z.string()).length(3),
      sentiment: z.enum(["positive", "negative", "neutral"]),
    }),
    prompt: `Summarise: ${text}`,
  });
  return Response.json(object);
}

Test with curl. Response is guaranteed-shaped — Zod validates.

Worked? JSON has exactly the three fields, with bullets array of length 3.

Tool calling

Create src/app/api/tools/route.ts

import { streamText, tool, stepCountIs } from "ai";
import { z } from "zod";

export async function POST(req: Request) {
  const { messages } = await req.json();
  const result = streamText({
    model: "google/gemini-2.5-flash-lite",
    messages,
    tools: {
      get_weather: tool({
        description: "Get the current weather for a city",
        inputSchema: z.object({ city: z.string() }),
        execute: async ({ city }) => ({ city, temp: 12, condition: "rain" }),
      }),
    },
    stopWhen: stepCountIs(5),
  });
  return result.toDataStreamResponse();
}

Test by sending "what's the weather in London?" — confirm the tool gets called.

Rate limiting + error handling

1. Sign up for Upstash Redis

Open upstash.com, sign up free, create a Redis database. Copy UPSTASH_REDIS_REST_URL and UPSTASH_REDIS_REST_TOKEN into .env.local.

2. Install + wire rate limit

npm install @upstash/ratelimit @upstash/redis

In src/app/api/chat/route.ts, add at the top:

import { Ratelimit } from "@upstash/ratelimit";
import { Redis } from "@upstash/redis";

const ratelimit = new Ratelimit({ redis: Redis.fromEnv(), limiter: Ratelimit.slidingWindow(10, "1 m") });

In the POST handler, before streamText, add:

const ip = req.headers.get("x-forwarded-for") ?? "anon";
const { success } = await ratelimit.limit(ip);
if (!success) return new Response("Too many requests", { status: 429 });
Worked? Hit the endpoint 11 times in a minute — the 11th gets 429.
