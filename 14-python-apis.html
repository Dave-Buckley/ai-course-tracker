2. 06-gather.py — concurrent fetches

import asyncio, time
import httpx

async def fetch(client, url):
    r = await client.get(url)
    return r.status_code

async def main():
    async with httpx.AsyncClient() as client:
        urls = ["https://httpbin.org/delay/1"] * 3
        start = time.time()
        results = await asyncio.gather(*(fetch(client, u) for u in urls))
        print(f"3 fetches in {time.time()-start:.2f}s ({results})")

asyncio.run(main())
Worked? 3 fetches complete in ~1s (concurrent), not ~3s.

Dependency injection

Add to app/main.py:

from fastapi import Depends, Header, HTTPException

def get_settings():
    return {"app_name": "AI Course"}

@app.get("/info")
def info(settings: dict = Depends(get_settings)):
    return settings

def verify_token(x_token: str = Header(...)):
    if x_token != "secret":
        raise HTTPException(status_code=401, detail="bad token")

@app.get("/protected")
def protected(_=Depends(verify_token)):
    return {"ok": True}

Test /protected — fails without header, succeeds with X-Token: secret.

Streaming responses

Add to app/main.py:

from fastapi.responses import StreamingResponse
import time

def slow_gen():
    for i in range(5):
        yield f"chunk {i}\n".encode()
        time.sleep(0.5)

@app.get("/stream")
def stream():
    return StreamingResponse(slow_gen(), media_type="text/plain")

import httpx
from typing import AsyncIterator

async def stream_summary(text: str, api_key: str) -> AsyncIterator[str]:
    url = f"https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-lite:streamGenerateContent?alt=sse&key={api_key}"
    payload = {"contents": [{"parts": [{"text": f"Summarise in 3 bullet points:\n\n{text}"}]}]}
    async with httpx.AsyncClient(timeout=60) as client:
        async with client.stream("POST", url, json=payload) as r:
            async for line in r.aiter_lines():
                if line.startswith("data: "):
                    yield line[6:] + "\n"

Mini-project: streaming endpoint + CORS

Create app/main.py

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import StreamingResponse
from pydantic_settings import BaseSettings
from app.schemas import SummariseRequest
from app.llm import stream_summary

class Settings(BaseSettings):
    GEMINI_API_KEY: str
    class Config: env_file = ".env"

settings = Settings()
app = FastAPI()
app.add_middleware(CORSMiddleware, allow_origins=["http://localhost:3000"], allow_methods=["*"], allow_headers=["*"])

@app.get("/health")
def health(): return {"status": "ok"}

@app.post("/summarise")
async def summarise(req: SummariseRequest):
    return StreamingResponse(
        stream_summary(req.text, settings.GEMINI_API_KEY),
        media_type="text/event-stream"
    )

Run + test:

curl --no-buffer -X POST http://localhost:8000/summarise -H "Content-Type: application/json" -d "{\"text\":\"Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua.\"}"
