Schema + seed data

1. Create three related tables

In the Neon SQL editor, run:

CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE posts (
  id SERIAL PRIMARY KEY,
  user_id INT REFERENCES users(id),
  title TEXT NOT NULL,
  body TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE tags (
  id SERIAL PRIMARY KEY,
  name TEXT UNIQUE NOT NULL
);

CREATE TABLE post_tags (
  post_id INT REFERENCES posts(id),
  tag_id INT REFERENCES tags(id),
  PRIMARY KEY (post_id, tag_id)
);

SELECT, WHERE, ORDER, LIMIT

Run each in Neon, save in 03-reads.sql

-- All users, just name + email
SELECT name, email FROM users;

-- Gmail users only, newest first, top 3
SELECT * FROM users
WHERE email LIKE '%@gmail.com'
ORDER BY created_at DESC
LIMIT 3;

-- Alice's recent posts
SELECT * FROM posts WHERE user_id = 1;

-- Distinct authors who have posted
SELECT DISTINCT user_id FROM posts;

Joins

Save in 04-joins.sql

-- INNER: users + their posts (only matched)
SELECT u.name, p.title
FROM users u INNER JOIN posts p ON p.user_id = u.id;

-- LEFT: all users, posts where they exist (NULL otherwise)
SELECT u.name, p.title
FROM users u LEFT JOIN posts p ON p.user_id = u.id;

-- 3-table: user, post title, tag names
SELECT u.name, p.title, t.name AS tag
FROM users u
JOIN posts p ON p.user_id = u.id
JOIN post_tags pt ON pt.post_id = p.id
JOIN tags t ON t.id = pt.tag_id;
Worked? The LEFT JOIN includes users with NULL post titles (users who haven't posted).

GROUP BY + aggregates

Save in 05-aggregates.sql

-- Posts per user
SELECT user_id, COUNT(*) AS post_count
FROM posts GROUP BY user_id;

-- Users with at least 2 posts (HAVING filters groups)
SELECT user_id, COUNT(*) FROM posts
GROUP BY user_id HAVING COUNT(*) >= 2;

-- Tag usage ranking
SELECT t.name, COUNT(*) AS uses
FROM tags t
JOIN post_tags pt ON pt.tag_id = t.id
GROUP BY t.name
ORDER BY uses DESC;

WHERE filters rows before grouping. HAVING filters groups after grouping.

Window functions

Save in 06-windows.sql

What: window functions compute over groups but keep all rows (unlike GROUP BY which collapses).

-- Number each user's posts newest first
SELECT title, user_id,
  ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY created_at DESC) AS rn
FROM posts;

-- Each row preserved, plus the user's total post count
SELECT user_id, title,
  COUNT(*) OVER (PARTITION BY user_id) AS user_total
FROM posts;

Postgres specifics: jsonb, arrays, indexes

Save in 07-postgres.sql

-- jsonb column
ALTER TABLE posts ADD COLUMN metadata JSONB DEFAULT '{}';
UPDATE posts SET metadata = '{"views": 100, "tags": ["a", "b"]}' WHERE id = 1;
SELECT id, metadata->>'views' AS views FROM posts WHERE metadata->>'views' = '100';

-- text array
ALTER TABLE posts ADD COLUMN keywords TEXT[];
UPDATE posts SET keywords = ARRAY['python', 'sql'] WHERE id = 1;
SELECT * FROM posts WHERE 'python' = ANY(keywords);

-- index for performance
CREATE INDEX idx_posts_user_id ON posts(user_id);
EXPLAIN ANALYZE SELECT * FROM posts WHERE user_id = 1;
Worked? The EXPLAIN output shows Index Scan using your new index.
