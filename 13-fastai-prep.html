Autograd — the magic

Why autograd mattersPyTorch tracks every operation on tensors with requires_grad=True in a graph. Calling .backward() walks that graph and fills in .grad on every parameter. This is what makes neural network training "just work".

Create 02-autograd.ipynb

import torch
x = torch.tensor(3.0, requires_grad=True)
y = x ** 2
y.backward()
print("dy/dx at x=3 =", x.grad)  # expect 6.0
# f(x) = x² + 2x + 1, f'(x) = 2x + 2
x.grad.zero_()  # reset
f = x**2 + 2*x + 1
f.backward()
print("f' at x=3 =", x.grad)  # expect 8.0
Worked? Both gradients match what the analytic derivative gives.

Linear regression by hand in PyTorch

Create 03-linreg.ipynb

import torch
torch.manual_seed(42)

X = torch.randn(100, 1)
y = 3 * X + 1 + 0.1 * torch.randn(100, 1)

w = torch.tensor([0.0], requires_grad=True)
b = torch.tensor([0.0], requires_grad=True)
lr = 0.05

for step in range(200):
    pred = X * w + b
    loss = ((pred - y) ** 2).mean()
    loss.backward()
    with torch.no_grad():
        w -= lr * w.grad
        b -= lr * b.grad
    w.grad.zero_(); b.grad.zero_()

print(f"learned w = {w.item():.3f}, b = {b.item():.3f}")
Worked? w ≈ 3, b ≈ 1.

nn.Module + nn.Linear

Create 04-nn-module.ipynb

import torch
import torch.nn as nn
import torch.optim as optim

torch.manual_seed(42)
X = torch.randn(100, 1)
y = 3 * X + 1 + 0.1 * torch.randn(100, 1)

class LinearReg(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc = nn.Linear(1, 1)
    def forward(self, x):
        return self.fc(x)

model = LinearReg()
loss_fn = nn.MSELoss()
opt = optim.SGD(model.parameters(), lr=0.05)

for step in range(200):
    opt.zero_grad()
    pred = model(X)
    loss = loss_fn(pred, y)
    loss.backward()
    opt.step()

w, b = model.fc.weight.item(), model.fc.bias.item()
print(f"learned w = {w:.3f}, b = {b:.3f}")
Worked? Same w ≈ 3, b ≈ 1, but cleaner code than block 4.

Datasets + DataLoaders

Create 05-dataset.ipynb

from torch.utils.data import Dataset, DataLoader
import torch

class FakeDataset(Dataset):
    def __init__(self, n=100):
        self.X = torch.randn(n, 1)
        self.y = 3 * self.X + 1 + 0.1 * torch.randn(n, 1)
    def __len__(self): return len(self.X)
    def __getitem__(self, i): return self.X[i], self.y[i]

ds = FakeDataset()
dl = DataLoader(ds, batch_size=8, shuffle=True)

for x_batch, y_batch in dl:
    print("batch shape:", x_batch.shape)
    break  # just show one
Worked? Prints torch.Size([8, 1]) — DataLoader gave a batch of 8.

Pick two visually-distinct classes you can grab 30+ images of each — e.g. "vinyl records vs CDs", "dogs vs squirrels", "your DJ gear vs PE gear".

2. Get images via duckduckgo_search

pip install duckduckgo_search
from duckduckgo_search import DDGS
from fastdownload import download_url
from pathlib import Path

def search(term, max_n=30):
    return [r["image"] for r in DDGS().images(term, max_results=max_n)]

for cls in ["vinyl record", "compact disc"]:
    folder = Path(f"images/{cls}")
    folder.mkdir(parents=True, exist_ok=True)
    for url in search(cls):
        try: download_url(url, folder, show_progress=False)
        except: pass
