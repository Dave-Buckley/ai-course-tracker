# Big-O cheat sheet

- O(1) — constant. Hash map lookup.
- O(log n) — halves each step. Binary search.
- O(n) — single pass. Loop over array.
- O(n log n) — sort.
- O(n²) — nested loops over same data.
- O(2ⁿ) — recursive without memoisation. Avoid.

Two-pointer pattern

When to reach for two pointersThe array is sorted (or you can sort it) and you're looking for a pair / triplet that satisfies a condition. Two indices walk toward each other from both ends.

1. 167-two-sum-ii.py — LeetCode 167

Open LeetCode 167, read the problem. In VS Code, new file 167-two-sum-ii.py. Type:

def two_sum(numbers: list[int], target: int) -> list[int]:
    left, right = 0, len(numbers) - 1
    while left < right:
        s = numbers[left] + numbers[right]
        if s == target:
            return [left + 1, right + 1]   # 1-indexed
        if s < target:
            left += 1
        else:
            right -= 1
    return []

# tests
assert two_sum([2, 7, 11, 15], 9) == [1, 2]
assert two_sum([2, 3, 4], 6) == [1, 3]
print("ok")

Hash maps — seen-set pattern

Signal"Find a pair that sums to X" or "is there a duplicate?" or "count frequency" → reach for a dict.

1. 01-two-sum.py — LeetCode 1

Create. Type:

def two_sum(nums: list[int], target: int) -> list[int]:
    seen = {}
    for i, n in enumerate(nums):
        if target - n in seen:
            return [seen[target - n], i]
        seen[n] = i
    return []

assert two_sum([2, 7, 11, 15], 9) == [0, 1]
print("ok")

Sliding window

Signal"Longest/max/min substring or subarray of size k" → expand a window from the right, shrink from the left.

1. 03-longest-substring.py — LeetCode 3

Create. Type:

def longest(s: str) -> int:
    seen = {}
    start = 0
    best = 0
    for end, c in enumerate(s):
        if c in seen and seen[c] >= start:
            start = seen[c] + 1
        seen[c] = end
        best = max(best, end - start + 1)
    return best

assert longest("abcabcbb") == 3
assert longest("bbbbb") == 1
print("ok")

from collections import defaultdict

def group_anagrams(strs: list[str]) -> list[list[str]]:
    groups = defaultdict(list)
    for s in strs:
        key = "".join(sorted(s))
        groups[key].append(s)
    return list(groups.values())

1. 206-reverse-list.py

Create. Type:

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val; self.next = next

def reverse(head: ListNode | None) -> ListNode | None:
    prev = None
    while head:
        nxt = head.next
        head.next = prev
        prev = head
        head = nxt
    return prev

Stacks and queues

1. 20-valid-parens.py — classic stack problem

Create. Type:

def is_valid(s: str) -> bool:
    stack = []
    pairs = {")": "(", "]": "[", "}": "{"}
    for c in s:
        if c in pairs:
            if not stack or stack.pop() != pairs[c]:
                return False
        else:
            stack.append(c)
    return not stack

assert is_valid("()[]{}") == True
assert is_valid("(]") == False
print("ok")

1. 104-max-depth.py

Create. Type:

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val; self.left = left; self.right = right

def max_depth(root: TreeNode | None) -> int:
    if not root: return 0
    return 1 + max(max_depth(root.left), max_depth(root.right))

Trees: BFS (level order)

1. 102-level-order.py

Create. Type:

from collections import deque

def level_order(root):
    if not root: return []
    result = []
    q = deque([root])
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        result.append(level)
    return result

1. 200-num-islands.py

Create. Type:

def num_islands(grid: list[list[str]]) -> int:
    if not grid: return 0
    rows, cols = len(grid), len(grid[0])
    seen = set()
    def dfs(r, c):
        if (r, c) in seen or r < 0 or c < 0 or r >= rows or c >= cols or grid[r][c] != "1":
            return
        seen.add((r, c))
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            dfs(r+dr, c+dc)
    count = 0
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == "1" and (r, c) not in seen:
                dfs(r, c); count += 1
    return count

Dynamic programming intro

1. 70-climbing-stairs.py — both versions

Create. Type both memoised and tabulated:

from functools import lru_cache

@lru_cache(None)
def climb_memo(n):
    if n <= 2: return n
    return climb_memo(n-1) + climb_memo(n-2)

def climb_tab(n):
    if n <= 2: return n
    a, b = 1, 2
    for _ in range(n - 2):
        a, b = b, a + b
    return b

print(climb_memo(10), climb_tab(10))
