Tools as functions

Create 01-tools-manual.py

import os, json
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

def add_numbers(a: float, b: float) -> float:
    return a + b

def get_weather(city: str) -> dict:
    return {"city": city, "temp_c": 12, "condition": "rain"}

# Define schemas for the LLM
tools = [
    genai.protos.Tool(function_declarations=[
        genai.protos.FunctionDeclaration(
            name="add_numbers",
            description="Add two numbers",
            parameters=genai.protos.Schema(
                type=genai.protos.Type.OBJECT,
                properties={"a": genai.protos.Schema(type=genai.protos.Type.NUMBER),
                           "b": genai.protos.Schema(type=genai.protos.Type.NUMBER)},
                required=["a", "b"],
            ),
        ),
        genai.protos.FunctionDeclaration(
            name="get_weather",
            description="Get weather for a city",
            parameters=genai.protos.Schema(
                type=genai.protos.Type.OBJECT,
                properties={"city": genai.protos.Schema(type=genai.protos.Type.STRING)},
                required=["city"],
            ),
        ),
    ])
]

model = genai.GenerativeModel("gemini-2.5-flash-lite", tools=tools)
chat = model.start_chat()
response = chat.send_message("What's 12 + 7? And what's the weather in London?")
for part in response.parts:
    if part.function_call:
        print("LLM wants to call:", part.function_call.name, dict(part.function_call.args))
Worked? The LLM emits two function-call requests, one per tool.

The agent loop

Create 02-agent-loop.py

import os, datetime
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

def add_numbers(a, b): return float(a) + float(b)
def get_weather(city): return {"city": city, "temp_c": 12, "condition": "rain"}
def current_time(): return datetime.datetime.utcnow().isoformat()

tool_fns = {"add_numbers": add_numbers, "get_weather": get_weather, "current_time": current_time}

# Use auto function-calling - simpler than block 2
model = genai.GenerativeModel("gemini-2.5-flash-lite", tools=list(tool_fns.values()))
chat = model.start_chat(enable_automatic_function_calling=True)
response = chat.send_message("What time is it now? Also tell me the weather in London.")
print(response.text)
Worked? Final answer mentions both the current time and London's weather.

Failure handling + iteration cap

Create 03-flaky.py

import random, os
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

def flaky_tool(query: str) -> str:
    """Sometimes fails, sometimes returns a result."""
    if random.random() < 0.5:
        return "ERROR: temporary failure, please retry once"
    return f"Result for {query}: success!"

model = genai.GenerativeModel("gemini-2.5-flash-lite", tools=[flaky_tool])
chat = model.start_chat(enable_automatic_function_calling=True)
response = chat.send_message("Call flaky_tool with query='hello'. If it fails, retry.")
print(response.text)
Worked? The LLM handles transient failure by retrying or pivoting.

Visible reasoning trace

Create 04-trace.py

import os
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

def square(n: float) -> float: return n * n
def double(n: float) -> float: return n * 2

model = genai.GenerativeModel("gemini-2.5-flash-lite", tools=[square, double])
chat = model.start_chat()
response = chat.send_message("What's the square of 7, then double it?")

for entry in chat.history:
    print(f"-- {entry.role} --")
    for part in entry.parts:
        if hasattr(part, "function_call") and part.function_call.name:
            print(f"   CALL: {part.function_call.name}({dict(part.function_call.args)})")
        elif hasattr(part, "function_response") and part.function_response.name:
            print(f"   RESULT: {dict(part.function_response.response)}")
        elif part.text:
            print(f"   TEXT: {part.text}")
Worked? You see a clear thought-action-observation chain.

Real tools: web search + fetch

Create 06-real-tools.py

import os
from duckduckgo_search import DDGS
import requests
from bs4 import BeautifulSoup
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

def web_search(query: str) -> list[dict]:
    """Search the web. Returns list of {title, url, snippet}."""
    return list(DDGS().text(query, max_results=3))

def fetch_page(url: str) -> str:
    """Fetch a URL and return cleaned text (first 3000 chars)."""
    html = requests.get(url, timeout=10).text
    text = BeautifulSoup(html, "html.parser").get_text(" ", strip=True)
    return text[:3000]

model = genai.GenerativeModel("gemini-2.5-flash-lite", tools=[web_search, fetch_page])
chat = model.start_chat(enable_automatic_function_calling=True)
response = chat.send_message("What did Anthropic publish about agents in 2024? Search, fetch the most relevant page, summarise.")
print(response.text)
Worked? Agent searches, picks a result, fetches, summarises in one go.

Research-assistant FastAPI agent

Create app/main.py

import os
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from duckduckgo_search import DDGS
import requests, google.generativeai as genai
from bs4 import BeautifulSoup
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

def web_search(query: str): return list(DDGS().text(query, max_results=3))
def fetch_page(url: str):
    html = requests.get(url, timeout=10).text
    return BeautifulSoup(html, "html.parser").get_text(" ", strip=True)[:3000]

model = genai.GenerativeModel("gemini-2.5-flash-lite", tools=[web_search, fetch_page])

class Q(BaseModel):
    question: str

app = FastAPI()

@app.post("/ask")
def ask(q: Q):
    chat = model.start_chat(enable_automatic_function_calling=True)
    response = chat.send_message(q.question, stream=True)
    def gen():
        for chunk in response:
            if chunk.text:
                yield chunk.text
    return StreamingResponse(gen(), media_type="text/plain")

@app.get("/health")
def health(): return {"status": "ok"}
