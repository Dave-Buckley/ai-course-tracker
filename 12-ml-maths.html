import numpy as np
import matplotlib.pyplot as plt

# f(x, y) = x² + y², gradient is [2x, 2y]
x, y = np.meshgrid(np.linspace(-3, 3, 20), np.linspace(-3, 3, 20))
gx = 2 * x; gy = 2 * y
plt.quiver(x, y, gx, gy)
plt.title("gradient of x² + y² points outward")
plt.axis("equal"); plt.show()
# Chain rule: d/dx sin(x²) = 2x cos(x²)
import sympy as sp
xs = sp.Symbol("x")
f = sp.sin(xs**2)
print("f =", f)
print("f' =", sp.diff(f, xs))

Gradient descent in action

Create 05-gradient-descent.ipynb

import numpy as np
import matplotlib.pyplot as plt

# minimise f(x) = (x - 2)² ; gradient = 2(x - 2)
def grad(x): return 2 * (x - 2)

x = 0.0
lr = 0.1
trajectory = [x]
for _ in range(50):
    x = x - lr * grad(x)
    trajectory.append(x)

plt.plot(trajectory)
plt.title(f"Converged to {x:.4f}")
plt.xlabel("iteration"); plt.ylabel("x"); plt.show()
# Try lr = 1.1 - watch it diverge
x = 0.0; lr = 1.1
traj = [x]
for _ in range(20):
    x = x - lr * grad(x); traj.append(x)
plt.plot(traj); plt.title("Too-large learning rate diverges"); plt.show()
Worked? First plot converges to ~2. Second blows up.

2-layer neural network in NumPy

No PyTorch, no fast.ai. Raw NumPy. The point is to feel every multiplication.

Create nn-from-scratch.ipynb

import numpy as np
import matplotlib.pyplot as plt

np.random.seed(42)

# Fake data: classify points by whether x+y > 0
X = np.random.randn(100, 2)
y = (X[:, 0] + X[:, 1] > 0).astype(int).reshape(-1, 1)

# 2 inputs -> 4 hidden (ReLU) -> 1 output (sigmoid)
W1 = np.random.randn(2, 4) * 0.5
b1 = np.zeros((1, 4))
W2 = np.random.randn(4, 1) * 0.5
b2 = np.zeros((1, 1))

def relu(x): return np.maximum(0, x)
def sigmoid(x): return 1 / (1 + np.exp(-x))

losses = []
lr = 0.1
for step in range(2000):
    # forward
    z1 = X @ W1 + b1
    a1 = relu(z1)
    z2 = a1 @ W2 + b2
    y_hat = sigmoid(z2)

    # binary cross-entropy
    eps = 1e-12
    loss = -np.mean(y * np.log(y_hat + eps) + (1 - y) * np.log(1 - y_hat + eps))
    losses.append(loss)

    # backward
    dL_dz2 = (y_hat - y) / len(X)
    dW2 = a1.T @ dL_dz2
    db2 = dL_dz2.sum(axis=0, keepdims=True)
    dL_da1 = dL_dz2 @ W2.T
    dL_dz1 = dL_da1 * (z1 > 0)
    dW1 = X.T @ dL_dz1
    db1 = dL_dz1.sum(axis=0, keepdims=True)

    W1 -= lr * dW1; b1 -= lr * db1
    W2 -= lr * dW2; b2 -= lr * db2

plt.plot(losses); plt.xlabel("iteration"); plt.ylabel("loss"); plt.show()
preds = (sigmoid(relu(X @ W1 + b1) @ W2 + b2) > 0.5).astype(int)
print(f"final accuracy: {(preds == y).mean():.4f}")
Worked? Loss curve decreases, final accuracy > 0.9.
